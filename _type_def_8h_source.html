199  template <typename T>
200  inline _binary<T> binary(T n) {
201  _binary<T> __binary;
202  constexpr int bits = _mypow(8U, sizeof(n)) - 1;
203 
204  for (int i = bits; i >= 0; --i) {
205  if ((n >> i) & 1) __binary._digits.append(1, '1');
206  else __binary._digits.append(1, '0');
207  }
208 
209  return __binary;
210  }
